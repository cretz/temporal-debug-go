 3import (

12func MyWorkflow(ctx workflow.Context) error {
13	ctx = workflow.WithActivityOptions(ctx, workflow.ActivityOptions{
14		StartToCloseTimeout: 30 * time.Minute,
15		HeartbeatTimeout:    5 * time.Second,
16		WaitForCancellation: true,
17	})
18	log := workflow.GetLogger(ctx)
19	log.Info("Cancel workflow started")
20
21	// Used to call activities by function pointer
22	var a *Activities
23	defer func() {
24
25		if !errors.Is(ctx.Err(), workflow.ErrCanceled) {
26			return
27		}
28
29		// When the workflow is canceled, it has to get a new disconnected context
30		newCtx, _ := workflow.NewDisconnectedContext(ctx)
31		err := workflow.ExecuteActivity(newCtx, a.CleanupActivity).Get(ctx, nil)
32		if err != nil {
33			log.Error("CleanupActivity failed", "Error", err)
34		}
35	}()
36
37	var result string
38	err := workflow.ExecuteActivity(ctx, a.ActivityToBeCanceled).Get(ctx, &result)
39	log.Info("ActivityToBeCanceled returned", "Result", result, "Error", result, err)
40
41	err = workflow.ExecuteActivity(ctx, a.ActivityToBeSkipped).Get(ctx, nil)
42	log.Error("Error from ActivityToBeSkipped", "Error", err)
43
44	log.Info("Workflow Execution complete")
45
46	return nil
47}
48
49type Activities struct{}
50
51func (a *Activities) ActivityToBeCanceled(ctx context.Context) (string, error) {
52	logger := activity.GetLogger(ctx)
53	logger.Info("activity started")
54	for {
55		select {
56		case <-time.After(1 * time.Second):
57			logger.Info("heartbeating...")
58			activity.RecordHeartbeat(ctx, "")
59		case <-ctx.Done():
60			logger.Info("context is cancelled")
61			return "I am canceled by Done", nil
62		}
63	}
64}
65
66func (a *Activities) CleanupActivity(ctx context.Context) error {
67	logger := activity.GetLogger(ctx)
68	logger.Info("Cleanup Activity started")
69	return nil
70}
71
72func (a *Activities) ActivityToBeSkipped(ctx context.Context) error {
73	logger := activity.GetLogger(ctx)
74	logger.Info("this Activity will be skipped due to cancellation")
75	return nil
76}
